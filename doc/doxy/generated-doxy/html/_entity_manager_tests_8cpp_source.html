140TEST_F(EntityManagerTests, DestroyEntityMany) {
141 using namespace GLESC::ECS;
142 EntityID createdEntities = 100;
143 // Destroyed entities must be less than created entities / 2
144 EntityID destroyedEntities = 35;
145 std::set<EntityID> destroyedEntitiesList;
146
147
148 for (EntityID entityId = firstEntityID; entityId < createdEntities + firstEntityID; ++entityId) {
149 entityManager.createNextEntity("TestEntity" + std::to_string(entityId), {});
150 }
151
152 // Destroy every second entity 1, 3, 5, 7, 9, ... to destroyedEntities * 2 (70)
153 for (EntityID entityId = firstEntityID; entityId < destroyedEntities * 2 + firstEntityID; entityId += 2) {
154 entityManager.destroyEntity(entityId);
155 destroyedEntitiesList.emplace(entityId);
156 }
157
158 TEST_SECTION("Checking interface checkers");
159 for (EntityID entityId : destroyedEntitiesList) {
160 // The destroyed entity does not exist by id
161 ASSERT_FALSE(getEntityManager().doesEntityExist(entityId));
162 // The destroyed entity does not exist by name
163 ASSERT_FALSE(getEntityManager().doesEntityExist("TestEntity" + std::to_string(entityId)));
164 // The destroyed entity is not alive
165 ASSERT_FALSE(getEntityManager().isEntityAlive(entityId));
166 // The destroyed entity can be created again
167 ASSERT_TRUE(getEntityManager().canEntityBeCreated("TestEntity" + std::to_string(entityId)));
168 }
169
170 TEST_SECTION("Checking entity manager internal data structures");
171 ASSERT_EQ(getEntityManager().getAvailableEntities().size(),
172 GLESC::ECS::maxEntities - (createdEntities - destroyedEntities));
173 // The entity IDs size is createdEntities - destroyedEntities
174 ASSERT_EQ(getEntityManager().getEntityNameToID().size(), createdEntities - destroyedEntities);
175 ASSERT_EQ(getEntityManager().getEntityIDToName().size(), createdEntities - destroyedEntities);
176 // The living entity count is createdEntities - destroyedEntities
177 ASSERT_EQ(getEntityManager().getLivingEntityCount(), createdEntities - destroyedEntities);
178 // Checking the the entity IDs contain the same entities we created
179 for (EntityID entityId = firstEntityID; entityId < createdEntities + firstEntityID; ++entityId) {
180 if (destroyedEntitiesList.find(entityId) != destroyedEntitiesList.end()) continue;
181 ASSERT_TRUE(getEntityManager().getEntityNameToID().at("TestEntity" + std::to_string(entityId)) == entityId);
182 ASSERT_TRUE(getEntityManager().getEntityIDToName().at(entityId) == "TestEntity" + std::to_string(entityId));
183 ASSERT_TRUE(getEntityManager().getSignatures().at(entityId).to_ulong() == 0);
184 }
185}
