173TEST(PlaneTests, IntersectsLine) {
174 using namespace GLESC::Math;
175
176 Plane plane(Direction(0, 0, 1), 0); // XY plane
177
178 // Line parallel to the plane, above the plane
179 Line parallelLineAbove(Point(0, 0, 1), Direction(1, 0, 0));
180 EXPECT_FALSE(plane.intersects(parallelLineAbove));
181
182 // Line parallel to the plane, below the plane
183 Line parallelLineBelow(Point(0, 0, -1), Direction(1, 0, 0));
184 EXPECT_FALSE(plane.intersects(parallelLineBelow));
185
186 // Line intersecting the plane
187 Line intersectingLine(Point(0, 0, -1), Direction(0, 0, 1));
188 EXPECT_TRUE(plane.intersects(intersectingLine));
189
190 // Line coincident with the plane
191 Line coincidentLine(Point(0, 0, 0), Direction(1, 0, 0));
192 EXPECT_TRUE(plane.intersects(coincidentLine));
193
194 // Line intersecting the plane at a non-origin point
195 Line nonOriginIntersectingLine(Point(1, 1, -1), Direction(0, 0, 1));
196 EXPECT_TRUE(plane.intersects(nonOriginIntersectingLine));
197
198 // Check intersection point
199 Point intersectionPoint;
200 EXPECT_TRUE(plane.intersects(intersectingLine, intersectionPoint));
201 EXPECT_EQ_VEC(intersectionPoint, Point(0, 0, 0)); // Expect intersection at the origin
202
203 // Check intersection point for non-origin intersecting line
204 EXPECT_TRUE(plane.intersects(nonOriginIntersectingLine, intersectionPoint));
205 EXPECT_EQ_VEC(intersectionPoint, Point(1, 1, 0)); // Expect intersection at (1, 1, 0)
206
207 // Line that starts on the plane and goes away
208 Line startsOnPlaneLine(Point(0, 0, 0), Direction(0, 1, 1));
209 EXPECT_TRUE(plane.intersects(startsOnPlaneLine));
210
211 // Line that starts on the plane and goes parallel
212 Line startsOnPlaneGoesParallel(Point(0, 0, 0), Direction(1, 0, 0));
213 EXPECT_TRUE(plane.intersects(startsOnPlaneGoesParallel));
214
215 // Test with a skewed line that should intersect
216 Line skewedIntersectingLine(Point(1, 2, 3), Direction(-4, -5, -6));
217 EXPECT_TRUE(plane.intersects(skewedIntersectingLine));
218
