89 Velocity getVelocityWithFriction(const Physics& physics, const std::vector<Physics*>& physicsOfCollided) {
90 Velocity velocityWithFriction = physics.getVelocity();
91 for (auto otherPhysics : physicsOfCollided) {
92 // Calculate the average friction coefficient between the two objects
93 float frictionCoefficient = otherPhysics->getFriction() + physics.getFriction();
94
95 // Calculate the friction force based on the velocity and friction coefficient
96 Vec3F frictionForce = physics.getVelocity() * frictionCoefficient;
97
98 // Apply the friction force to the total velocity
99 velocityWithFriction -= frictionForce;
100 }

115 void PhysicsManager::handleCollisions(Collider& collider, Physics& physics) const {
116 CollisionInformation& info = collider.getCollisionInformation();
117 executeCallbacks(collider, info);
118
119 // If the object is static, no new velocity is needed, it will remain 0
120 if (physics.isStatic()) return;
121
122 // If the velocity is too small, we consider it as 0 so we don't have jittering
123 bool bouncingOccurs = !info.wasOnGroundLastFrame() &&
124 physics.getVelocity().length() > bouncinessThreshold;
125
126 Velocity finalVelocity;
127 if (bouncingOccurs) {
128 // Calculate the collision normal to reflect the velocity
129 Vec3F collisionNormal = {
130 info.getCollidingAxis().getX(), info.getCollidingAxis().getY(), info.getCollidingAxis().getZ()
131 };
132 finalVelocity = getBouncingVelocity(physics.getVelocity(), collisionNormal, bounciness);
133 }
134 else {
135 AxisConstraints freeAxis = getAxisFree(collider);
136 bool frictionApplied = isFrictionApplied(physics, freeAxis);
137
138 Vec3F velocityWithFriction;
139 if (frictionApplied)
140 velocityWithFriction = physics.getVelocity();
141 else
142 velocityWithFriction= getVelocityWithFriction(physics, info.getPhysicsOfCollided());
143
144 for (int i = 0; i < 3; ++i) {
145 if (Math::gt(velocityWithFriction[i], 0.0f))
146 velocityWithFriction[i] *= static_cast<float>(freeAxis[i][0]);
147 else
148 velocityWithFriction[i] *= static_cast<float>(freeAxis[i][1]);
149 }
150 finalVelocity = velocityWithFriction;
151 }
152
153
154 physics.setVelocity(finalVelocity);
