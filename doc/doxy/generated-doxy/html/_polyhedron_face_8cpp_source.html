28[[nodiscard]] bool PolyhedronFace::intersects(const PolyhedronFace &face) const {
29 D_ASSERT_FALSE(polyhedronVertices.empty(), "Polyhedron vertices must not be empty.");
30 // Get normals of both faces
31 const Direction& thisNormal = getNormal();
32 const Direction& otherNormal = face.getNormal();
33
34 // Vector to store all axes to test
35 std::vector<Direction> axes;
36
37 // Add face normals to axes (to handle face-face parallel cases)
38 axes.push_back(thisNormal);
39 axes.push_back(otherNormal);
40
41 // Function to add unique axes from edge cross products
42 auto addUniqueAxis = [&axes](const Direction &edgeNormal) {
43 // Normalize the edge normal
44 Direction normalized = edgeNormal.normalize();
45 // Check if this normal is unique
46 for (const auto &axis : axes) {
47 if (axis.isParallel(normalized)) {
48 return; // Skip if parallel axis already exists
49 }
50 }
51 axes.push_back(normalized); // Add unique axis
52 };
53
54 // Generate axes from edges of this face
55 for (size_t i = 0; i < indices.size(); ++i) {
56 size_t next = (i + 1) % indices.size();
57 Direction edge = polyhedronVertices.at(next) - polyhedronVertices.at(i);
58 addUniqueAxis(thisNormal.cross(edge));
59 }
60
61 // Generate axes from edges of other face
62 for (size_t i = 0; i < face.indices.size(); ++i) {
63 size_t next = (i + 1) % face.indices.size();
64 Direction edge = face.polyhedronVertices.at(next) - face.polyhedronVertices.at(i);
65 addUniqueAxis(otherNormal.cross(edge));
66 }
67
68 // Check for overlap on all axes
69 for (const auto &axis : axes) {
70 double thisMin = std::numeric_limits<double>::max();
71 double thisMax = std::numeric_limits<double>::lowest();
72 double otherMin = std::numeric_limits<double>::max();
73 double otherMax = std::numeric_limits<double>::lowest();
74
75 // Project vertices of both faces onto the axis
76 for (auto index : indices) {
77 double projection = polyhedronVertices.at(index).dot(axis);
78 thisMin = min(thisMin, projection);
79 thisMax = max(thisMax, projection);
80 }
81 for (auto index : face.indices) {
82 double projection = face.polyhedronVertices.at(index).dot(axis);
83 otherMin = min(otherMin, projection);
84 otherMax = max(otherMax, projection);
85 }
86
87 // Check for separation on this axis
88 if (thisMax < otherMin || otherMax < thisMin) {
89 return false; // No overlap, faces do not intersect
90 }
91 }
92
93 return true; // Overlap on all axes, faces intersect
94}

97bool PolyhedronFace::intersects(const Point &point) const {
98 D_ASSERT_FALSE(polyhedronVertices.empty(), "Polyhedron vertices must not be empty.");
99 for (size_t i = 0; i < polyhedronVertices.size(); ++i) {
100 Direction edge = polyhedronVertices.at((i + 1) % polyhedronVertices.size()) - polyhedronVertices.at(i);
101 Direction toPoint = point - polyhedronVertices.at(i);
102 if (getNormal().cross(edge).dot(toPoint) < 0) {
103 return false; // Point is outside this edge
104 }
105 }
106 return true; // Point is inside all edges
107}

109bool PolyhedronFace::intersects(const Line &line) const {
110 D_ASSERT_FALSE(polyhedronVertices.empty(), "Polyhedron vertices must not be empty.");
111 // Check if the line is parallel to the plane
112 Distance dotProduct = getNormal().dot(line.getDirection());
113 if (Math::eq(Math::abs(dotProduct),0)) {
114 return false; // Line is parallel to the plane, no intersection
115 }
116
117 // Find the intersection point with the plane
118 float t = -(getNormal().dot(line.getPoint() - polyhedronVertices.at(0)) + plane.getDistance() / dotProduct);
119 Direction intersectionPoint = line.getPoint() + line.getDirection() * t;
120
121 // Check if the intersection point is inside the face
122 return intersects(intersectionPoint);
123}

125bool PolyhedronFace::intersects(const Plane &planeParam) const {
126 D_ASSERT_FALSE(polyhedronVertices.empty(), "Polyhedron vertices must not be empty.");
127 // Counters for vertices on each side of the plane and on the plane
128 int positive = 0, negative = 0, onPlane = 0;
129
130 for (auto index : indices) {
131 // Calculate the signed distance from the vertex to the plane
132 Distance distance = planeParam.distanceToPoint(polyhedronVertices.at(index));
133
134 const Distance intersectsEpsilon = 0.0001;
135
136 // Count vertices on each side of the plane and on the plane
137 if(eq(distance, 0, intersectsEpsilon)) onPlane++;
138 else if (distance > 0) positive++;
139 else if (distance < 0) negative++;
140
141
142 // Check for early exit conditions
143 if (positive > 0 && negative > 0) return true; // Vertices on both sides
144 if (onPlane > 0 && (positive > 0 || negative > 0)) return true; // An edge or vertex on the plane
145 }
146
147 // Check if any edge lies entirely on the plane
148 if (onPlane >= 2) return true;
149
150 // No intersection found
151 return false;
152}
