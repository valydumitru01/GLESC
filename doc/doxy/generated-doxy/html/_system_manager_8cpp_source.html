32void SystemManager::entitySignatureChanged(EntityID entity, Signature entitySignature) {
33 D_ASSERT_FALSE(systemSignatures.empty(),
34 "All systems must be registered before entities can be associated with them");
35 // Notify each system that an entity's signature changed
36 for (auto& [name, entitySet] : associatedEntities) {
37 if ((entitySignature & systemSignatures[name]) == systemSignatures[name]) {
38 // If the systemSignature of the entity matches the systemSignature of the system, insert it into the set
39 entitySet.insert(entity);
40 }
41 else {
42 entitySet.erase(entity);
43 }
44 }
45}
