43SDL_SurfacePtr TextureLoader::flipSurface(SDL_Surface &surface) {
44 SDL_SurfacePtr flippedSurface = std::unique_ptr<SDL_Surface, decltype(&SDL_FreeSurface)>(
45 SDL_CreateRGBSurface(0, surface.w, surface.h, surface.format->BitsPerPixel,
46 surface.format->Rmask, surface.format->Gmask,
47 surface.format->Bmask, surface.format->Amask),
48 &SDL_FreeSurface);
49
50 D_ASSERT_NOT_NULLPTR(flippedSurface, "Flipped surface couldn't be created. SDL error: " +
51 std::string(SDL_GetError()));
52
53 Logger::get().info("Flipping image...");
54
55 D_ASSERT_TRUE(!SDL_MUSTLOCK(&surface) || SDL_LockSurface(&surface) == 0,
56 "Original surface must be locked.");
57
58 D_ASSERT_TRUE(!SDL_MUSTLOCK(flippedSurface.get()) || SDL_LockSurface(flippedSurface.get()) == 0,
59 "Flipped surface must be locked.");
60
61
62 int pitch = surface.pitch;
63 for (int currentLine = 0; currentLine < surface.h; ++currentLine) {
64 memcpy(static_cast<unsigned char *>(flippedSurface->pixels) + currentLine * pitch,
65 static_cast<unsigned char *>(surface.pixels) + (surface.h - 1 - currentLine) * pitch,
66 pitch);
67 }
68
69 if (SDL_MUSTLOCK(&surface)) {
70 SDL_UnlockSurface(&surface);
71 }
72 if (SDL_MUSTLOCK(flippedSurface.get())) {
73 SDL_UnlockSurface(flippedSurface.get());
74 }
75
76 Logger::get().success("Image flipped successfully.");
77
78 return flippedSurface;
