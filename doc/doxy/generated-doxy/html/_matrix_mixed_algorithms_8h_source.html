54 template <typename TypeMat, typename TypeEye, typename TypeTarget, typename TypeUp>
55 static void lookAt3D(const VectorData<TypeEye, 3>& eye,
56 const VectorData<TypeTarget, 3>& target,
57 const VectorData<TypeUp, 3>& up,
58 MatrixData<TypeMat, 4, 4>& resMatrix) {
59 VectorData<TypeMat, 3> f;
60 VectorAlgorithms::vectorSub(target, eye, f); // forward = target - eye
61 VectorAlgorithms::normalize(f, f);
62
63 VectorData<TypeMat, 3> s; // s = right
64 VectorAlgorithms::crossProduct(f, up, s);
65 VectorAlgorithms::normalize(s, s);
66
67 VectorData<TypeMat, 3> u; // u = up
68 VectorAlgorithms::crossProduct(s, f, u);
69
70 resMatrix = {
71 {
72 {s[0], u[0], -f[0], 0},
73 {s[1], u[1], -f[1], 0},
74 {s[2], u[2], -f[2], 0},
75 {
76 -VectorAlgorithms::dotProduct(s, eye),
77 -VectorAlgorithms::dotProduct(u, eye),
78 VectorAlgorithms::dotProduct(f, eye),
79 1
80 }
81 }

90 template <typename TypeMat, typename Type>
91 static void matrixToEulerRotation(const MatrixData<TypeMat, 4, 4>& mat, VectorData<Type, 3>& result) {
92 Type pitch = std::asin(-mat[2][1]); // Calculate pitch
93
94 Type cosPitch = std::cos(pitch);
95 Type yaw, roll;
96
97 if (std::abs(cosPitch) > 1e-6) {
98 // Gimbal lock check
99 yaw = std::atan2(mat[2][0], mat[2][2]);
100 roll = std::atan2(mat[0][1], mat[1][1]);
101 }
102 else {
103 yaw = std::atan2(-mat[0][2], mat[0][0]);
104 roll = 0;
105 }
106
107 result = {pitch, yaw, roll};
108 }

120 template <typename TypeDgrs, typename TypeRes>
121 static void getRotate3DMatrixForAxis(TypeDgrs rads,
122 const VectorData<TypeDgrs, 3>& axisVec,
123 MatrixData<TypeRes, 4, 4>& result) {
124 const TypeRes c = Math::cos(rads);
125 const TypeRes s = Math::sin(rads);
126 const TypeRes t = TypeRes(1) - c;
127
128 // Normalized axis vector
129 VectorData<TypeDgrs, 3> axis;
130 VectorAlgorithms::normalize(axisVec, axis);
131
132 VectorData<TypeRes, 3> temp;
133 VectorAlgorithms::vectorScalarMul(axis, t, temp);
134
135 result[0][0] = c + temp[0] * axis[0];
136 result[0][1] = temp[0] * axis[1] + s * axis[2];
137 result[0][2] = temp[0] * axis[2] - s * axis[1];
138 result[0][3] = 0;
139
140 result[1][0] = temp[1] * axis[0] - s * axis[2];
141 result[1][1] = c + temp[1] * axis[1];
142 result[1][2] = temp[1] * axis[2] + s * axis[0];
143 result[1][3] = 0;
144
145 result[2][0] = temp[2] * axis[0] + s * axis[1];
146 result[2][1] = temp[2] * axis[1] - s * axis[0];
147 result[2][2] = c + temp[2] * axis[2];
148 result[2][3] = 0;
149
150 result[3][0] = 0;
151 result[3][1] = 0;
152 result[3][2] = 0;
153 result[3][3] = 1;
154 }
