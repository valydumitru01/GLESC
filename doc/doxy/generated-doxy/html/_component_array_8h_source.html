95 void removeData(EntityID entity) {
96 PRINT_COMPONENT_ARRAY_STATUS("Before removing data from entity " + std::to_string(entity));
97 D_ASSERT_TRUE(hasComponent(entity), "Entity does not have component");
98
99 // Copy element at end into deleted element's place to maintain density
100 size_t indexOfRemovedEntity = entityToIndexMap[entity];
101 size_t indexOfLastElement = size - 1;
102 // Call destructor on removed component, it is not called automatically causing bugs and memory leaks
103 componentArray[indexOfRemovedEntity].~Component();
104 new(&componentArray[indexOfRemovedEntity]) Component(std::move(componentArray[indexOfLastElement]));
105
106 // Update map to point to moved spot
107 EntityID entityOfLastElement = indexToEntityMap[indexOfLastElement];
108
109 entityToIndexMap[entityOfLastElement] = indexOfRemovedEntity;
110 indexToEntityMap[indexOfRemovedEntity] = entityOfLastElement;
111
112 // Erase the entity from both maps
113 entityToIndexMap.erase(entity);
114 indexToEntityMap.erase(indexOfLastElement);
115 --size;
116
117 PRINT_COMPONENT_ARRAY_STATUS("After removing data from entity " + std::to_string(entity));
118 }
