50void printBufferData(const GLESC::GAPI::Void *data, GLESC::GAPI::Size size) {
51 const auto *typedData = static_cast<const BufferType *>(data);
52 size_t count = size / sizeof(BufferType); // Calculate how many elements are in the buffer
53
54 // Determine the maximum width needed for alignment
55 BufferType max_value = *std::max_element(typedData, typedData + count);
56 int max_width = std::to_string(max_value).length();
57
58 for (size_t i = 0; i < count; ++i) {
59 // Use std::setw to set the width and std::setfill to fill with spaces
60 std::cout << std::setw(max_width) << std::setfill(' ') << typedData[i] << " ";
61 if (i % 8 == 7) std::cout << std::endl; // Print 8 elements per line for readability
62 }
63 std::cout << std::endl;
64}
