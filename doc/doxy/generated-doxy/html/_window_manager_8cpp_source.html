140uint32_t WindowManager::getRaisedFlags() {
141 // Flags that are needed to be passed to the windowManager * to configure it.
142 // To add more flags we need to use the binary OR ( | )
143 // More info: https://wiki.libsdl.org/SDL_WindowFlags
144
145 uint32_t flags = 0;
146#if GLESC_RENDER_API == OpenGLAPI
147 // Flag to allow SDL windowManager work with OpenGL
148 flags |= SDL_WINDOW_OPENGL;
149#elif GLESC_RENDER_API == VulkanAPI
150 flags |= SDL_WINDOW_VULKAN;
151#endif
152 // Window has no borders
153 // flags |= SDL_WINDOW_BORDERLESS;
154 // Window grabs input focus
155 // flags |= SDL_WINDOW_INPUT_GRABBED;
156 // Window enables High pixel density if supported by monitor
157 flags |= SDL_WINDOW_ALLOW_HIGHDPI;
158 // Window is resizable
159 flags |= SDL_WINDOW_RESIZABLE;
160
161 return flags;
162}

173SDL_Window* WindowManager::createWindow(const char* title) {
174 uint32_t flags = getRaisedFlags();
175 SDL_Window* tempWindow = SDL_CreateWindow(title, GLESC_WINDOW_X, GLESC_WINDOW_Y,
176 GLESC_WINDOW_WIDTH, GLESC_WINDOW_HEIGHT, flags);
177 postSDLCall;
178 D_ASSERT_NOT_NULLPTR(tempWindow, "Unable to create windowManager: " + std::string(SDL_GetError()));
179 Logger::get().success("Window created!");
180 SDLCall(SDL_SetWindowMinimumSize(tempWindow, windowMinWidth, windowMinHeight));
181
182 return tempWindow;
183}
