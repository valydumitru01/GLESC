44void HUDManager::removeWindow(WindowID id) {
45 auto it = windowIDs.find(id);
46 if (it != windowIDs.end()) {
47 // Remove from vector
48 auto win_it = std::find(windows.begin(), windows.end(), it->second);
49 if (win_it != windows.end()) {
50 windows.erase(win_it);
51 }
52 // Remove from map
53 windowIDs.erase(it);
54 } else {
55 D_ASSERT_TRUE(false, "Window ID not found");
56 }
