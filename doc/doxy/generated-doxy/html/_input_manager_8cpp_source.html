27void InputManager::update(bool& running) {
28 SDL_Point sdlMousePos = {0, 0};
29 SDL_Event event;
30 while (SDL_PollEvent(&event)) {
31 hudManager.processInput(event);
32 switch (event.type) {
33 case SDL_QUIT:
34 running = false;
35 break;
36 case SDL_WINDOWEVENT:
37 handleWindowEvent(event);
38 break;
39 case SDL_KEYDOWN:
40 case SDL_KEYUP:
41 handleKeyEvent(event);
42 break;
43 case SDL_MOUSEBUTTONDOWN:
44 case SDL_MOUSEBUTTONUP:
45 handleMouseButton(event);
46 break;
47 case SDL_MOUSEMOTION:
48 handleMouseMotion(event);
49 if (SDL_GetRelativeMouseMode() == SDL_TRUE) {
50 sdlMousePos = {event.motion.xrel, event.motion.yrel};
51 }
52 break;
53 default:
54 break;
55 }
56 }
57 // When no input is given to the mouse, send a 0,0 movement to the buffer and lerp the mouse position
58 if (SDL_GetRelativeMouseMode() == SDL_TRUE) {
59 addToBuffer(sdlMousePos.x, sdlMousePos.y);
60 lerpMousePos(mousePos, getAvgMousePos());
61 }
62}

127MousePosition InputManager::getAvgMousePos() const {

133 if (!deltaBuffer.empty()) {
134 float avgX = sumX / static_cast<float>(deltaBuffer.size());

152bool InputManager::checkKeyAction(const KeyInput& keyInput) {
153 auto it = keyMap.find(keyInput.key);
154 if (it != keyMap.end()) {
155 auto& state = it->second;
156 switch (keyInput.action) {
157 case KeyAction::ONGOING_PRESSED:
158 return state.pressed;
159 case KeyAction::ONCE_PRESSED:
160 if (state.justPressed) {
161 state.justPressed = false; // Reset the flag
162 return true;
163 }
164 break;
165 case KeyAction::ONCE_RELEASED:
166 if (state.justReleased) {
167 state.justReleased = false; // Reset the flag
168 return true;
169 }
170 break;
171 default:
172 break;
173 }
174 }
175 return false;
176}
