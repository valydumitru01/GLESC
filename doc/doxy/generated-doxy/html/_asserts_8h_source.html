82template <typename Type1, typename Type2>
83bool assertEqualsEq(const Type1& value, const Type2& expected) {
84 // Floating point comparison
85 if constexpr (std::is_floating_point_v<Type1> || std::is_floating_point_v<Type2>) {
86 using CommonType = std::common_type_t<Type1, Type2>;
87 const CommonType epsilon = std::numeric_limits<CommonType>::epsilon();
88 const CommonType diff = std::fabs(static_cast<CommonType>(value) - static_cast<CommonType>(expected));
89 return diff <= epsilon * std::max(std::fabs(static_cast<CommonType>(value)),
90 std::fabs(static_cast<CommonType>(expected)))
91 || diff < epsilon;
92 }
93 // Pointer value comparison
94 else if constexpr (std::is_pointer_v<Type1> && std::is_pointer_v<Type2>) {
95 return reinterpret_cast<const void*>(value) == reinterpret_cast<const void*>(expected);
96 }
97 // General comparison
98 else {
99 return value == expected;
100 }
101}

104#define ASSERT_CONTENT(failureCondition, message) \
105 if (!(static_cast<bool>(failureCondition))) { \
106 std::ostringstream oss; \
107 oss << "\n==================== ASSERTION FAILED ====================\n" \
108 << "Failed Condition: " << #failureCondition << "\n" \
109 << "Message : " << (message) << "\n\n" \
110 << "Location : " << __FILE__ << ", Line " << __LINE__ << "\n" \
111 << "Function : " << __PRETTY_FUNCTION__ << "\n\n"; \
112 throw AssertFailedException(oss.str()); \
113 }
