87void Skybox::calculateAverageColor() {
88 // To calculate the average color of the skybox we'll get 20 pixels at random
89 // and calculate the average color of those pixels
90 // (technicall its gonna be 120 as we get 6 pixels per sample)
91 for (int i = 0; i < PIXEL_SAMPLE_COUNT; i++) {
92 // Its a cube, with the width is enough, no need more checking
93 Size pixelWidth = skyboxCubemap.getCubemapTextures()[0].getWidth();
94 Size x = Math::generateRandomNumber(static_cast<Size>(0), pixelWidth - 1);
95 Size y = Math::generateRandomNumber(static_cast<Size>(0), pixelWidth - 1);
96 std::array<Pixel, 6> pixels = skyboxCubemap.getPixelsAtCoords(x, y);
97 ColorRgb finalColor;
98 for (int pix = 0; pix < 6; pix++) {
99 ColorRgb pixColor =
100 ColorRgb(pixels[pix].r, pixels[pix].g, pixels[pix].b);
101 finalColor += pixColor;
102 }
103 finalColor /= pixels.size();
104 averageColor += finalColor;
105 }
106 averageColor /= PIXEL_SAMPLE_COUNT;
107}
