10ColorMesh MeshFactory::sphere(const int numSlices, const int numStacks, const float radius, const ColorRgba& color) {
11 ColorMesh mesh;
12 mesh.startBuilding();
13
14 constexpr auto piF = Math::pi<float>();
15 // Create the top and bottom cap
16 for (int i = 0; i < numSlices; ++i) {
17 float theta1 = 2 * piF * static_cast<float>(i) / static_cast<float>(numSlices);
18 float theta2 = 2 * piF * static_cast<float>(i + 1) / static_cast<float>(numSlices);
19 const auto stacks = static_cast<float>(numStacks);
20
21 Position topP1(Math::cos(theta2) * Math::sin(piF / stacks),
22 Math::cos(piF / stacks),
23 Math::sin(theta2) * Math::sin(piF / stacks));
24 Position topP2(Math::cos(theta1) * Math::sin(piF / stacks),
25 Math::cos(piF / stacks),
26 Math::sin(theta1) * Math::sin(piF / stacks));
27
28 Position bottomP1(Math::cos(theta2) * Math::sin(piF - piF / stacks),
29 Math::cos(piF - piF / stacks),
30 Math::sin(theta2) * Math::sin(piF - piF / stacks));
31 Position bottomP2(Math::cos(theta1) * Math::sin(piF - piF / stacks),
32 Math::cos(piF - piF / stacks),
33 Math::sin(theta1) * Math::sin(piF - piF / stacks));
34 // Top cap triangles
35 mesh.addTris(
36 {Position(0, radius, 0), color},
37 {topP1 * radius, color},
38 {topP2 * radius, color}
39 );
40
41 // Bottom cap triangles
42 mesh.addTris(
43 {Position(0, -radius, 0), color},
44 {bottomP2 * radius, color},
45 {bottomP1 * radius, color}
46 );
47 }
48
49 // Create the middle quads
50 for (int j = 1; j < numStacks - 1; ++j) {
51 for (int i = 0; i < numSlices; ++i) {
52 float theta1 = 2 * piF * static_cast<float>(i) / static_cast<float>(numSlices);
53 float theta2 = 2 * piF * static_cast<float>(i + 1) / static_cast<float>(numSlices);
54 float phi1 = piF * static_cast<float>(j) / static_cast<float>(numStacks);
55 float phi2 = piF * static_cast<float>(j + 1) / static_cast<float>(numStacks);
56
57 Position p1(Math::cos(theta1) * Math::sin(phi1), Math::cos(phi1), Math::sin(theta1) * Math::sin(phi1));
58 Position p2(Math::cos(theta2) * Math::sin(phi1), Math::cos(phi1), Math::sin(theta2) * Math::sin(phi1));
59 Position p3(Math::cos(theta2) * Math::sin(phi2), Math::cos(phi2), Math::sin(theta2) * Math::sin(phi2));
60 Position p4(Math::cos(theta1) * Math::sin(phi2), Math::cos(phi2), Math::sin(theta1) * Math::sin(phi2));
61
62
63 mesh.addQuad(
64 {p1 * radius, color},
65 {p2 * radius, color},
66 {p3 * radius, color},
67 {p4 * radius, color}
68 );
69 }
70 }
71 mesh.finishBuilding();
72 return mesh;
73}

114 ColorMesh mesh;

116
117 double w = width / 2.0, h = height, d = depth / 2.0;
118
119 // Base
120 mesh.addQuad(

122 {Position(w, 0, -d), color},
123 {Position(w, 0, d), color},
124 {Position(-w, 0, d), color}
125 );
126
127 // Sides
128 Position top(0, h, 0);

130 Position(-w, 0, -d),
131 Position(w, 0, -d),
132 Position(w, 0, d),
133 Position(-w, 0, d)
134 };
135
136 for (int i = 0; i < 4; ++i) {
137 mesh.addTris(
138 {top, color},
139 {corners[i], color},
140 {corners[(i + 1) % 4], color}
141 );
142 }
143 mesh.finishBuilding();
144 return mesh;
