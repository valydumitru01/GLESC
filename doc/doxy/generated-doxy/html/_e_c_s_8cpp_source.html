70const std::set<EntityID>& ECSCoordinator::getAssociatedEntities(const SystemName& name) const {
71 if (!systemManager.isSystemRegistered(name))
72 return {};
73 const auto& set = systemManager.getAssociatedEntitiesOfSystem(name);
