94[[nodiscard]] bool Plane::intersects(const Plane& plane) const {
95 bool isParallel = normal.isParallel(plane.normal);
96 // When the dot product is 1, the planes are parallel
97 // So we check if the dot product is not 1, which means the planes are not parallel
98 return !isParallel ||
99 // And if they are parallel we check if the distance between the planes is 0,
100 // which means the planes are the same, so they intersect
101 (eq(distance, plane.distance) && isParallel);
102}

104[[nodiscard]] bool Plane::intersects(const Line& line) const {
105 // Check if the line direction is parallel to the plane normal
106 // This is determined by checking if their dot product is close to zero
107 double dotProduct = normal.dot(line.getDirection());
108 bool isParallel = eq(dotProduct, 0);
109
110 if (isParallel) {
111 // If the line is parallel to the plane, check if any point on the line lies in the plane
112 // Use the distance formula: d = (n â‹… p) + D
113 // If the distance is close to 0, the line lies in the plane
114 return eq(abs(distanceToPoint(line.getPoint())), 0);
115 }
116 // If the line is not parallel to the plane, it will intersect the plane at some point
117 return true;
118}
