251
252 // Set a window title with the entity's name

257 // Dynamically calculate the needed height for each child window based on the number of items
258 float childHeight = 15 * ImGui::GetTextLineHeightWithSpacing();
259
260 // Begin a child window with a specific height to enforce stacking
261 ImGui::BeginChild(stat.name.c_str(), ImVec2(-1, childHeight), true);
262
263 // Set a title for the component inside its frame
264 ImGui::Text("%s", stat.name.c_str());
265 ImGui::Separator();
266
267 // Display each attribute of the component
268 for (EntityStatsManager::Value& value : *stat.values) {
269 if (value.isString) {
270 ImGui::Text(value.name.c_str());
271 ImGui::Text(value.stringData.c_str());
272 continue;
273 }
274 // While visible the panel of the data, the value is dirty as we want to update it
275 if (value.valueDirty != nullptr)
276 *value.valueDirty = true;
277 putInputForType(value.type, value);
278 }
279 for (const EntityStatsManager::Value& value : stat.getUpdatedValues()) {
280 ImGui::Text(value.name.c_str());
281 ImGui::Text(value.stringData.c_str());
282 }
283 ImGui::EndChild(); // End of component frame
284 ImGui::Spacing(); // Add some space between components for better readability
285 }
286 ImGui::PopFont();
