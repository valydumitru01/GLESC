70void Engine::update() {
71 Logger::get().importantInfoWhite("Engine update started");
72
73 hudManager.update();
74 game.update();
75 for (ECS::EntityID id : ecs.getEntitiesToBeDestroyed()) {
76#ifndef NDEBUG_GLESC
77 EntityListManager::entityRemoved(ecs.getEntityName(id));
78#endif
79 if (ecs.hasComponent<ECS::RenderComponent>(id) && ecs.hasComponent<ECS::TransformComponent>(id)) {
80 renderer.remove(ecs.getComponent<ECS::RenderComponent>(id).getMesh(),
81 ecs.getComponent<ECS::TransformComponent>(id).transform);
82 }
83 }
84 ecs.destroyEntities();
85#ifndef NDEBUG_GLESC
86 // We need to clear the hud items (Sun, Fog, etc) here, if not called here, juttering will occur
87 // Or we will get memory leaks
88 HudItemsManager::clearItems();
89#endif
90
91
92 for (auto& system : systems) {
93 system->update();
94 }
95 // This tells the renderer that all the data it needs to render has been updated
96 // (Update and render are decoupled, therefore not necesarily consecutive)
97 renderer.setRendererUpdated();
98 for (const auto& [name,id] : ecs.getAllEntities()) {
99 if (ecs.isEntityInstanced(name))
100 if ((engineCamera.getEntity().getComponent<ECS::TransformComponent>().transform.getPosition().distance(
101 ecs.getComponent<ECS::TransformComponent>(id).transform.getPosition()) > 10000.0f)) {
102 ecs.markForDestruction(id);
103 }
104 }
105 Console::log("Debug log message");
106 Console::warn("Debug warning message");
107 Console::error("Debug error message");
108
109 Logger::get().importantInfoWhite("Engine update finished");
110}

112std::vector<std::unique_ptr<ECS::System>> Engine::createSystems() {
113 std::vector<std::unique_ptr<ECS::System>> systems;
114 systems.push_back(std::make_unique<ECS::RenderSystem>(renderer, ecs));
115 systems.push_back(std::make_unique<ECS::TransformSystem>(ecs));
116 // Physics system must update before the physics collision system
117 systems.push_back(std::make_unique<ECS::PhysicsSystem>(physicsManager, ecs));
118 systems.push_back(std::make_unique<ECS::PhysicsCollisionSystem>(physicsManager, collisionManager, ecs));
119 systems.push_back(std::make_unique<ECS::InputSystem>(inputManager, ecs));
120 systems.push_back(std::make_unique<ECS::CameraSystem>(renderer, windowManager, ecs));
121 systems.push_back(std::make_unique<ECS::LightSystem>(ecs, renderer));
122 systems.push_back(std::make_unique<ECS::SunSystem>(ecs, renderer));
123 systems.push_back(std::make_unique<ECS::FogSystem>(renderer, ecs));
124#ifndef NDEBUG_GLESC
125 systems.push_back(std::make_unique<ECS::DebugInfoSystem>(ecs, renderer));
126#endif
127 return systems;
128}
