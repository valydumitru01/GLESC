18EntityID EntityManager::createNextEntity(const EntityName& nameParam, const EntityMetadata& metadata) {
19 EntityID id = availableEntities.front();
20
21 EntityName name = nameParam;
22 if (metadata.type == EntityType::Instance) {
23 instancedEntities[nameParam].push_back(id);
24 name.append(std::to_string(instancesNextID[nameParam]));
25 instancesNextID[nameParam]++;
26 }
27
28
29 D_ASSERT_TRUE(canEntityBeCreated(name), "Entity must be able to be created");
30
31 availableEntities.pop();
32 entityIDToName.insert({id, name});
33 entityNameToID.insert({name, id});
34 entityMetadata.insert({id, metadata});
35 ++livingEntityCount;
36 ++entityCounter;
37
38 D_ASSERT_TRUE(doesEntityExist(id), "Entity must exist after creation");
39 return id;
40}

42void EntityManager::destroyEntity(EntityID entity) {
43 D_ASSERT_TRUE(areThereLivingEntities(), "There must be living entities to destroy one");
44 D_ASSERT_TRUE(doesEntityExist(entity), "Entity must exist before destruction");
45 D_ASSERT_TRUE((entity != GLESC::ECS::EntityManager::nullEntity), "Entity must not be nullEntity");
46
47 if (isEntityInstanced(entityIDToName[entity])) {
48 EntityName instanceName = Stringer::strip(entityIDToName[entity], "1234567890");
49 std::vector<EntityID>& instancedList = this->instancedEntities[instanceName];
50
51 auto it = std::find(instancedList.begin(), instancedList.end(), entity);
52 if (it != instancedList.end())
53 instancedList.erase(it);
54 }
55
56
57 signatures[entity].reset();
58 availableEntities.push(entity);
59 entityNameToID.erase(entityIDToName[entity]);
60 entityIDToName.erase(entity);
61 entityMetadata.erase(entity);
62
63
64 --livingEntityCount;
65
66 D_ASSERT_FALSE(doesEntityExist(entity), "Entity must not exist after destruction");
67 D_ASSERT_FALSE(isEntityAlive(entity), "Entity must not be alive after destruction");
68}
