20void TextureSurface::initializeData(SDL_Surface& sdlSurface) {
21 // TODO: Is all this copying really necessary? Maybe use the SDL_Surface directly?
22 // But having our own format is useful
23
24 format = Format();
25
26 format.rMask = sdlSurface.format->Rmask;
27 format.gMask = sdlSurface.format->Gmask;
28 format.bMask = sdlSurface.format->Bmask;
29 format.aMask = sdlSurface.format->Amask;
30 format.rLoss = sdlSurface.format->Rloss;
31 format.gLoss = sdlSurface.format->Gloss;
32 format.bLoss = sdlSurface.format->Bloss;
33 format.aLoss = sdlSurface.format->Aloss;
34 format.rShift = sdlSurface.format->Rshift;
35 format.gShift = sdlSurface.format->Gshift;
36 format.bShift = sdlSurface.format->Bshift;
37 format.aShift = sdlSurface.format->Ashift;
38
39
40 // Assert that the bits per pixel is as expected
41 D_ASSERT_TRUE(isDppCorrect(sdlSurface.format->BitsPerPixel), "Bits per pixel is not correct");
42
43 if (format.aMask)
44 format.colorFormat = ColorFormat::RGBA;
45 else
46 format.colorFormat = ColorFormat::RGB;
47
48 // Set format's bit depth
49 format.bitDepth = static_cast<BitDepth>(sdlSurface.format->BitsPerPixel);
50
51 // Set texture dimensions
52 width = static_cast<UInt>(sdlSurface.w);
53 height = static_cast<UInt>(sdlSurface.h);
54
55 // Calculate the total number of bytes for all pixels
56 size_t totalBytes = width * height * static_cast<size_t>(sdlSurface.format->BytesPerPixel);
57
58 // Resize the pixel vector to accommodate all bytes
59 pixels.resize(totalBytes);
60
61 // Copy the pixel data using memcpy for efficiency
62 std::memcpy(pixels.data(), sdlSurface.pixels, totalBytes);
63}

65Pixel TextureSurface::getPixel(unsigned int x, unsigned int y) const {
66 D_ASSERT_TRUE(hasLoaded, "Texture not loaded");
67 D_ASSERT_LESS(x, width, "X coordinate out of bounds");
68 D_ASSERT_LESS(y, height, "Y coordinate out of bounds");
69 auto baseUBytePixelPtr = pixels.data();
70
71 // Calculate the number of bytes per pixel
72 // Divide by 8 because the bit depth is stored in bits and each byte has 8 bits
73 auto byteDepth = static_cast<UInt>(format.bitDepth) / 8;
74
75 // Move the pointer to the pixel we want to read
76 // Multiply by byteDepth because the pixel data is stored as bytes
77 baseUBytePixelPtr += (y * width + x) * byteDepth;
78 Uint32 pixelValue = 0;
79
80 for (UInt i = 0; i < byteDepth; ++i) {
81 pixelValue |= static_cast<Uint32>(baseUBytePixelPtr[i]) << (8 * i);
82 }
83
84 // Extract the individual color components
85 auto r = static_cast<Uint8>((pixelValue & format.rMask) >> format.rShift);
86 auto g = static_cast<Uint8>((pixelValue & format.gMask) >> format.gShift);
87 auto b = static_cast<Uint8>((pixelValue & format.bMask) >> format.bShift);
88 Uint8 a = (format.aMask != 0)
89 ? static_cast<Uint8>((pixelValue & format.aMask) >> format.aShift)
90 : 255;
91
92 // Normalize if needed (i.e. if the bit depth is less than 8)
93 r = (format.rLoss != 0) ? r * 255 / ((1 << format.rLoss) - 1) : r;
94 g = (format.gLoss != 0) ? g * 255 / ((1 << format.gLoss) - 1) : g;
95 b = (format.bLoss != 0) ? b * 255 / ((1 << format.bLoss) - 1) : b;
96 // Alpha normalization is typically not needed if alpha is either 0 or 255
97
98 return Pixel{r, g, b, a};
99}

119std::string TextureSurface::toString() const {

131 unsigned int heightStep = std::max(1u, getHeight() / pixelCount);
132 Logger::get().info("\tPixels:");

134 for (UInt x = 0; x < getWidth() - 1; x += widthStep) {
135 auto pixel = getPixel(x, y);
136
137 Logger::get().info("\t\tPixel at (" + std::to_string(x) + ", " + std::to_string(y) +
138 "): " + pixel.toString());
139 }
140 }
