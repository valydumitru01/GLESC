433void textMatrixWithGlm() {
434 if constexpr (GLESC::Math::MatrixAlgorithms::columnMajorMatrix) {
435 using MyMatLeft = GLESC::Math::Matrix<Type, M, N>;
436 using MyMatRight = GLESC::Math::Matrix<Type, X, M>;
437 using MyMatResult = GLESC::Math::Matrix<Type, X, N>;
438
439 using GlmMatLeft = glm::mat<M, N, Type>;
440 using GlmMatRight = glm::mat<X, M, Type>;
441 using GlmMatResult = glm::mat<X, N, Type>;
442
443 MyMatLeft matrixToMultiplyLeft;
444 initializeMatrixWithValues(matrixToMultiplyLeft);
445
446 MyMatRight matrixToMultiplyRight;
447 initializeMatrixWithValues(matrixToMultiplyRight);
448
449 MyMatResult actualMultResult;
450
451 actualMultResult = matrixToMultiplyLeft * matrixToMultiplyRight;
452
453 GlmMatLeft glmMatrixToMultiplyLeft{};
454 for (int i = 0; i < M; ++i)
455 for (int j = 0; j < N; ++j)
456 glmMatrixToMultiplyLeft[i][j] = matrixToMultiplyLeft[i][j];
457 GlmMatRight glmMatrixToMultiplyRight{};
458 for (int i = 0; i < X; ++i)
459 for (int j = 0; j < M; ++j)
460 glmMatrixToMultiplyRight[i][j] = matrixToMultiplyRight[i][j];
461
462 MyMatResult expectedMultResult;
463 GlmMatResult glmExpectedMultResult{};
464 glmExpectedMultResult = glmMatrixToMultiplyLeft * glmMatrixToMultiplyRight;
465 for (int i = 0; i < X; ++i)
466 for (int j = 0; j < N; ++j)
467 expectedMultResult[i][j] = glmExpectedMultResult[i][j];
468
469 EXPECT_EQ_MAT(actualMultResult, expectedMultResult);
470 }
471 else {
472 using MyMatLeft = GLESC::Math::Matrix<Type, N, M>;
473 using MyMatRight = GLESC::Math::Matrix<Type, M, X>;
474 using MyMatResult = GLESC::Math::Matrix<Type, N, X>;
475
476 using GlmMatLeft = glm::mat<N, M, Type>;
477 using GlmMatRight = glm::mat<M, X, Type>;
478 using GlmMatResult = glm::mat<N, X, Type>;

480
481 MyMatLeft matrixToMultiplyLeft;
482 initializeMatrixWithValues(matrixToMultiplyLeft);
483
484 MyMatRight matrixToMultiplyRight;
