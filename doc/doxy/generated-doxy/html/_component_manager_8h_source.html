195 void ComponentManager::registerComponent() {
196 S_ASSERT_TRUE((std::is_base_of_v<IComponent, Component>), "Component must inherit from IComponent");
197 D_ASSERT_TRUE(!isComponentRegistered<Component>(), "Component is already registered");
198
199 const char* typeName = typeid(Component).name();
200 componentArrays.try_emplace(typeName,
201 std::make_shared<ComponentArray<Component>>());
202 componentIDs.insert({typeName, nextComponentID});
203 componentNames.insert({nextComponentID, typeName});
204
205 ++nextComponentID;
206 }
