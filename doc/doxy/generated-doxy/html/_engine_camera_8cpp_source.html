58void EngineCamera::rotate(const MousePosition& deltaMouse) {
59 if (!inputManager.isMouseRelative()) return;
60
61
62 auto& transformComp = camera.getComponent<ECS::TransformComponent>().transform;
63 float cameraSensitivityValue = cameraSensitivity.get();
64
65 // Adjust the pitch based on mouse Y movement (up/down)
66 float mouseAdditionPitch = -deltaMouse.getY() * cameraSensitivityValue;
67 float nextMousePitch = transformComp.getRotation().getX() + mouseAdditionPitch;
68
69 // Only apply pitch rotation if it is within the constraints
70 if (nextMousePitch < cameraPitchConstraint &&
71 nextMousePitch > -cameraPitchConstraint) {
72 transformComp.setRotation(Transform::RotationAxis::Pitch, nextMousePitch);
73 }
74
75 // Adjust the yaw based on mouse X movement (left/right)
76 float mouseAdditionYaw = -deltaMouse.getX() * cameraSensitivityValue;
77 transformComp.addRotation(Transform::RotationAxis::Yaw, mouseAdditionYaw);
78}
