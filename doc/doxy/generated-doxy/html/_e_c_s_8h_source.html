285 template <typename Component>
286 void ECSCoordinator::addComponentRequirementToSystem(const SystemName& name) {
287 std::lock_guard<std::shared_mutex> lock(ecsMutex);
288 if (!systemManager.isSystemRegistered(name)) {
289 Logger::get().warning("System " + name + " is not registered. Cannot add component requirement.");
290 return;
291 }
292 PRINT_ECS_STATUS(
293 std::string("Before adding component") + typeid(Component).name() + std::string(" to system ") + name);
294 componentManager.registerComponentIfNotRegistered<Component>();
295 systemManager.addComponentRequirementToSystem(name, componentManager.getComponentID<Component>());
296 PRINT_ECS_STATUS(
297 std::string("After adding component") + typeid(Component).name() + std::string(" to system ") + name);
298 }
