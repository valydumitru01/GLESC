59void ShaderLoader::extractShaderCode(const std::string& shaderSource,
60 std::string& vertexCode,
61 std::string& fragmentCode) {
62
63 size_t vertexPos = shaderSource.find(tokenVertex);
64 size_t fragmentPos = shaderSource.find(tokenFragment);
65
66 validateShaderTokens(vertexPos, fragmentPos);
67
68 // Extract shader codes
69 if (vertexPos < fragmentPos) {
70 vertexCode = shaderSource.substr(vertexPos + strlen(tokenVertex),
71 fragmentPos - vertexPos - strlen(tokenVertex));
72 fragmentCode = shaderSource.substr(fragmentPos + strlen(tokenFragment));
73 }
74 else {
75 fragmentCode = shaderSource.substr(fragmentPos + strlen(tokenFragment),
76 vertexPos - fragmentPos - strlen(tokenFragment));
77 vertexCode = shaderSource.substr(vertexPos + strlen(tokenVertex));
78 }
79
80 validateShaderCodes(vertexCode, fragmentCode);
81}
