295TEST_F(ECSTests, CreateAndDeleteManyEntitiesAlternating) {
296 int entities = 100;
297 std::vector<GLESC::ECS::EntityID> createdEntities;
298 // First creating half of maxEntities
299 for (int i = 0; i < entities / 2; ++i) {
300 ecs.createEntity("TestEntity" + std::to_string(i), {});
301 createdEntities.push_back(i);
302 }
303 ASSERT_EQ(getEntityManager().getLivingEntityCount(), entities / 2);
304 ASSERT_EQ(getEntityManager().getEntityNameToID().size(), entities / 2);
305 ASSERT_EQ(getEntityManager().getEntityIDToName().size(), entities / 2);
306 ASSERT_EQ(getEntityManager().getAvailableEntities().size(), GLESC::ECS::maxEntities - entities / 2);
307
308 // Destroying one third of the entities, leaving 1/2 - 1/3 = 1/6 of the entities
309 for (int i = 0; i < entities / 3; ++i) {
310 ecs.destroyEntity(createdEntities.front());
311 createdEntities.erase(createdEntities.begin());
312 }
313 ASSERT_EQ(getEntityManager().getLivingEntityCount(), entities / 2 - entities / 3);
314 ASSERT_EQ(getEntityManager().getEntityNameToID().size(), entities / 2 - entities / 3);
315 ASSERT_EQ(getEntityManager().getEntityIDToName().size(), entities / 2 - entities / 3);
316 ASSERT_EQ(getEntityManager().getAvailableEntities().size(), GLESC::ECS::maxEntities - entities / 2 + entities / 3);
317
318 // Creating 1/3 of the entities, leaving 1/6 + 1/3 = 1/2 of the entities
319 for (int i = 0; i < entities / 3; ++i) {
320 ecs.createEntity("TestEntity" + std::to_string(i + entities / 2), {}); // Ensure unique names
321 createdEntities.push_back(i + entities / 2);
322 }
323 ASSERT_EQ(getEntityManager().getLivingEntityCount(), entities / 2);
324 ASSERT_EQ(getEntityManager().getEntityNameToID().size(), entities / 2);
325 ASSERT_EQ(getEntityManager().getEntityIDToName().size(), entities / 2);
326 ASSERT_EQ(getEntityManager().getAvailableEntities().size(), GLESC::ECS::maxEntities - entities / 2);
327
328 // Destroying the rest of the entities
329 for (int i = 0; i < entities / 2; ++i) {
330 ecs.destroyEntity(createdEntities.front());
331 createdEntities.erase(createdEntities.begin());
332 }
333 ASSERT_EQ(getEntityManager().getLivingEntityCount(), 0);
334 ASSERT_EQ(getEntityManager().getEntityNameToID().size(), 0);
335 ASSERT_EQ(getEntityManager().getEntityIDToName().size(), 0);
336 ASSERT_EQ(getEntityManager().getAvailableEntities().size(), GLESC::ECS::maxEntities);
