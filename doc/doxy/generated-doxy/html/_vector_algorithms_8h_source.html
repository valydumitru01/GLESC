567 template <typename Type>
568 static void vectorToEuler(const VectorData<Type, 3>& direction, const VectorData<Type, 3>& up,
569 VectorData<Type, 3>& rotation) {
570 VectorData<Type, 3> normalizedDirection;
571 VectorAlgorithms::normalize(direction, normalizedDirection);
572
573 VectorData<Type, 3> right;
574 VectorAlgorithms::crossProduct(normalizedDirection, up, right);
575 VectorAlgorithms::normalize(right, right);
576
577 VectorData<Type, 3> newUp;
578 VectorAlgorithms::crossProduct(right, normalizedDirection, newUp);
579
580 Type pitch, yaw, roll;
581
582 // Calculate yaw (rotation around the y-axis)
583 if (!Math::eq(normalizedDirection[0], Type(0)) || !Math::eq(normalizedDirection[2], Type(0))) {
584 yaw = Math::atan2(normalizedDirection[0], normalizedDirection[2]);
585 }
586 else {
587 yaw = 0;
588 }
589
590 // Calculate pitch (rotation around the x-axis)
591 Type horizontalDist = Math::sqrt(
592 normalizedDirection[0] * normalizedDirection[0] + normalizedDirection[2] * normalizedDirection[2]);
593 pitch = Math::atan2(-normalizedDirection[1], horizontalDist);
594
595 // Calculate roll (rotation around the z-axis)
596 if (!Math::eq(up[0], Type(0)) || !Math::eq(up[1], Type(0)) || !Math::eq(up[2], Type(0))) {
597 roll = Math::atan2(newUp[0], newUp[1]);
598 }
599 else {
600 roll = 0;
601 }
602
603 rotation[0] = pitch;
604 rotation[1] = yaw;
605 rotation[2] = roll;
606 }

699 template <typename Type, size_t N>
700 static void makeUpVector(const VectorData<Type, N>& rotationRads, VectorData<Type, N>& result) {
701 Type pitch = rotationRads[0];
702 Type yaw = rotationRads[1];
703 Type roll = rotationRads[2];

720 template <typename Type>
721 static void rotateVector3D(const VectorData<Type, 3>& vec,
722 const VectorData<Type, 3>& rotationRads,
723 VectorData<Type, 3>& result) {
724 Type cosX = Math::cos(rotationRads[0]);
725 Type sinX = Math::sin(rotationRads[0]);
726 Type cosY = Math::cos(rotationRads[1]);
727 Type sinY = Math::sin(rotationRads[1]);
728 Type cosZ = Math::cos(rotationRads[2]);
729 Type sinZ = Math::sin(rotationRads[2]);
730
731 // Rotation matrix for X axis
732 VectorData<Type, 3> rotatedX = {
733 vec[0],
734 vec[1] * cosX - vec[2] * sinX,
735 vec[1] * sinX + vec[2] * cosX
736 };
737
738 // Rotation matrix for Y axis
739 VectorData<Type, 3> rotatedY = {
740 rotatedX[0] * cosY + rotatedX[2] * sinY,
741 rotatedX[1],
742 -rotatedX[0] * sinY + rotatedX[2] * cosY
743 };
744
745 // Rotation matrix for Z axis
746 result = {
747 rotatedY[0] * cosZ - rotatedY[1] * sinZ,
748 rotatedY[0] * sinZ + rotatedY[1] * cosZ,
749 rotatedY[2]
750 };
751 }

778 template <typename Type, size_t N>
779 static void getOrthogonal(const VectorData<Type, N>& data, VectorData<Type, N>& result) {
780 static_assert(N == 2 || N == 3 || N == 4, "getOrthogonal is only implemented for 2D, 3D, and 4D vectors");
781
782 if constexpr (N == 2) {
783 // In 2D, the orthogonal vector can be obtained by swapping the data
784 // and negating one of them.
785 result = {-data[1], data[0]};
786 }
787 else if constexpr (N == 3) {
788 // In 3D, we use the cross product to find an orthogonal vector.
789 // First, we need a non-parallel vector to cross with.
790 // We choose either (0, 1, 0) or (1, 0, 0) depending on which one is less parallel
791 // to the original vector, to avoid degenerate cross products.
792 VectorData<Type, N> nonParallel;
793 if (Math::abs(data[0]) > Math::abs(data[1]))
794 nonParallel = VectorData<Type, N>({0, 1, 0});
795 else
796 nonParallel = VectorData<Type, N>({1, 0, 0});
797
798 // The cross product of this vector with 'nonParallel' yields an orthogonal vector.
799 VectorAlgorithms::crossProduct(data, nonParallel, result);
800 }
801 else if constexpr (N == 4) {
802 // Finding an orthogonal vector in 4D involves more choices due to extra degrees of
803 // freedom. Here, we zero out two data and solve the other two to maintain
804 // orthogonality. The choice of which data to zero out depends on the
805 // non-zero data of the original vector.
806 if (data[0] != 0 || data[1] != 0) {
807 // If the first or second component is non-zero, we set the last two data
808 // to zero and negate one of the first two data to create orthogonality.
809 result = VectorData<Type, N>({-data[1], data[0], 0, 0});
810 }
811 else {
812 // If the first two data are zero, we work with the last two data.
813 // We negate one and copy the other to create orthogonality.
814 result = VectorData<Type, N>({0, -data[3], data[2], 0});
815 }
816 }

828 template <typename Type, size_t N>
829 static bool isMultipleOf(const VectorData<Type, N>& vec1, const VectorData<Type, N>& vec2) {
830 Type ratio;
831 bool ratioInitialized = false;
832
833 for (size_t i = 0; i < N; ++i) {
834 if (!Math::eq(vec2[i], Type(0))) {
835 if (!ratioInitialized) {
836 ratio = vec1[i] / vec2[i];
837 ratioInitialized = true;
838 }
839 else {
840 if (!Math::eq(vec1[i] / vec2[i], ratio)) {
841 return false;
842 }
843 }
844 }
845 else if (!Math::eq(vec1[i], Type(0))) {
846 return false;
847 }
848 }
849
850 return true;
851 }
