214 void addTris(const VertexColorParam& a, const VertexColorParam& b, const VertexColorParam& c) {
215 D_ASSERT_TRUE(isBuilding, "Mesh is not being built");
216 Normal normal = calculateNormal(a.position, b.position, c.position);
217 unsigned int v1Index;
218 unsigned int v2Index;
219 unsigned int v3Index;
220 if constexpr (std::is_same_v<Vertex, ColorVertex>) {
221 v1Index = this->addVertex({a.position, normal, a.color});
222 v2Index = this->addVertex({b.position, normal, b.color});
223 v3Index = this->addVertex({c.position, normal, c.color});

225 else {

227 v2Index = this->addVertex({b.position, normal, b.textureCoordinate});
228 v3Index = this->addVertex({c.position, normal, c.textureCoordinate});

230 this->addTris(v1Index, v2Index, v3Index);

357 void addTris(Index index1, Index index2, Index index3) {
358 D_ASSERT_TRUE(isBuilding, "Mesh is not being built");
359 D_ASSERT_NOT_EQUAL(index1, index2, "Index 1 and 2 are equal");
360 D_ASSERT_NOT_EQUAL(index1, index3, "Index 1 and 3 are equal");
361 D_ASSERT_NOT_EQUAL(index2, index3, "Index 2 and 3 are equal");
362 D_ASSERT_LESS(index1, vertices.size(), "Index 1 out of bounds");
363 D_ASSERT_LESS(index2, vertices.size(), "Index 2 out of bounds");
364 D_ASSERT_LESS(index3, vertices.size(), "Index 3 out of bounds");
365 {
366 std::lock_guard<std::mutex> lock(indicesMutex);
367 indices.push_back(index1);
368 indices.push_back(index2);
369 indices.push_back(index3);
370 faces.push_back({index1, index2, index3});
371 }
372 this->dirtyFlag = true;

470 void sendToGpuBuffers() const {
471 if (wasDataSentToGpu) return;
472 vertexArray = std::make_unique<GLESC::GAPI::VertexArray>();
473
474 const void* bufferData = getVertices().data();
475 size_t bufferCount = getVertices().size();
476 size_t vertexBytes = sizeof(Vertex);
477 GLESC::GAPI::Enums::BufferUsages bufferUsage = getBufferUsage(getRenderType());
478
479 vertexArray->bind();
480
481 vertexBuffer = std::make_unique<GLESC::GAPI::VertexBuffer>(
482 bufferData,
483 bufferCount,
484 vertexBytes,
485 bufferUsage
486 );
487
488 indexBuffer = std::make_unique<GLESC::GAPI::IndexBuffer>(
489 getIndices().data(),
490 getIndices().size());
491
492 GLESC::GAPI::VertexBufferLayout layout;
493 for (GLESC::GAPI::Enums::Types type : getVertexLayout()) {
494 layout.push(type);
495 }
496
497 vertexArray->addBuffer(*vertexBuffer, layout);
498 wasDataSentToGpu = true;
499 }

550 size_t calculateHash() const {
551 size_t hashValue = 0;
552
553 // Hash vertex data
554 for (const auto& vertex : vertices) {
555 Hasher::hashCombine(hashValue, std::hash<Vertex>{}(vertex));
556 }
557 // Hash index data

559 Hasher::hashCombine(hashValue, std::hash<Index>{}(index));
560 }
