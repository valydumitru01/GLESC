44void SoundPlayer::play3DSound(const std::string& name,
45 const Vec3F& listenerPos,
46 const Vec3F& soundPos) {
47 // Calculate the distance between the listener and the sound
48 Vec3F diff = soundPos - listenerPos;
49 float distance = diff.length();
50
51 // Assuming maxDistance is the distance at which the sound is inaudible
52 const float maxDistance = 100.0f;
53 float volume = (maxDistance - distance) / maxDistance;
54 volume = std::max(0.0f, std::min(1.0f, volume));
55
56 // Calculate the panning based on the relative position on the x-axis
57 float pan = diff.getX() / maxDistance;
58 pan = std::max(-1.0f, std::min(1.0f, pan));
59
60 // Convert volume and pan to SDL_mixer's format
61 int sdlVolume = static_cast<int>(volume * MIX_MAX_VOLUME);
62 int left = static_cast<int>((1.0f - pan) * static_cast<float>(sdlVolume));
63 int right = static_cast<int>((1.0f + pan) * static_cast<float>(sdlVolume));
64
65 Channel channel = playSound(name, sdlVolume);
66 Mix_SetPanning(channel, left, right);
67}
