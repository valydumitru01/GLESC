123Position Transformer::transformVector(const Position& position, const Render::Model& matrix) {
124 // IMPORTANT! We use row major matrices but the data distribution is prepared to be column major for in GPU
125 // operations. So for the CPU we need to transpose the matrices.
126
127 Vec4F transformedToWorldVertex = matrix * position.homogenize();
128 return transformedToWorldVertex.dehomogenize();
129}

156void Interpolator::pushTransform(const Transform& transform) {
157 // Check if the transform has changed significantly
158 if (Math::abs(transform.getPosition().length() - lastTransform.getPosition().length()) > 100.f) {
159 lastTransform = transform;

167Transform Interpolator::interpolate(float alphaParam) const {
168 // If rotation difference is more than 180 degrees, don't interpolate
169 if (Math::abs(lastTransform.getRotation().getX() - currentTransform.getRotation().getX()) > 180.0f ||
170 Math::abs(lastTransform.getRotation().getY() - currentTransform.getRotation().getY()) > 180.0f ||
171 Math::abs(lastTransform.getRotation().getZ() - currentTransform.getRotation().getZ()) > 180.0f) {
172 return currentTransform;
173 }
174 // Clamp alpha to [0, 1]
175 float alpha = Math::min(alphaParam, 1.0f);
176 Transform interpolatedTransform;
177 interpolatedTransform.setPosition(
178 lastTransform.getPosition().lerp(currentTransform.getPosition(), alpha));
179 interpolatedTransform.setRotation(
180 lastTransform.getRotation().lerp(currentTransform.getRotation(), alpha));
181 interpolatedTransform.setScale(
182 lastTransform.getScale().lerp(currentTransform.getScale(), alpha));
183 return interpolatedTransform;
184}
