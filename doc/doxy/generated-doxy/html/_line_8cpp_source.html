28Distance Line::distance(const Point& pointParam) const {
29 // Calculate the cross product vector
30 auto crossProduct = direction.cross(pointParam - this->point);
31
32 // Return the length of the cross product vector,
33 // which represents the area of the parallelogram formed by the two vectors
34 // and is equal to the distance times the length of the direction vector
35 // Because the direction vector is normalized, its length is 1, and the length of the cross product
36 // is the distance from the point to the line.
37 return crossProduct.length();
38}
