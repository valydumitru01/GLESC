21void HudLookAndFeel::addFont(const std::string& fileName, FontSize size) {
22 validateFontName(fileName);
23 validateFontSize(size);
24 std::string fullFontPath = BinPath::getExecutableDirectory() + ASSETS_PATH + FONTS_PATH_FROM_ASSETS + fileName +
25 FONTS_EXTENSION;
26 //fullFontPath= GLESC::Stringer::replace(fullFontPath, "/", "\\");
27 Logger::get().info("Loading font: " + fullFontPath);
28 ImGuiIO& io = ImGui::GetIO();
29 ImFont* font = io.Fonts->AddFontFromFileTTF(fullFontPath.c_str(), static_cast<float>(size));
30 D_ASSERT_NOT_NULLPTR(font, "Failed to load font: " + fullFontPath);
31 fonts[fileName + std::to_string(size)] = font;
32}

66ImFont* HudLookAndFeel::getFont(const std::string& fileName, FontSize size) {
67 validateFontName(fileName);
68 validateFontSize(size);
69 if (fonts.find(fileName + std::to_string(size)) == fonts.end()) {
70 addFont(fileName, size);
71 }
72 return fonts[fileName + std::to_string(size)];
