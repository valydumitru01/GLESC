141ImVec2 InGameWindow::calculatePosition(ImVec2 windowSize) const {
142 float screenWidth = ImGui::GetIO().DisplaySize.x;
143 float screenHeight = ImGui::GetIO().DisplaySize.y;
144
145 // Calculate position based on fractions of screen width and height, correctly centering the window
146 float posX = (screenWidth * positionFraction.x) - (windowSize.x * center.x);
147 float posY = (screenHeight * positionFraction.y) - (windowSize.y * center.y);
148
149 // Ensure the position respects the screen boundaries
150 posX = std::max(posX, windowMarginFraction * screenWidth);
151 posY = std::max(posY, windowMarginFraction * screenHeight);
152 posX = std::min(posX, screenWidth - windowSize.x - windowMarginFraction * screenWidth);
153 posY = std::min(posY, screenHeight - windowSize.y - windowMarginFraction * screenHeight);
154
155 return {posX, posY};
156}

158ImVec2 InGameWindow::calculateSize() const {
159 float screenWidth = ImGui::GetIO().DisplaySize.x;
160 float screenHeight = ImGui::GetIO().DisplaySize.y;
161
162 // Calculate size based on fractions of screen width and height
163 ImVec2 calculatedSize = ImVec2(screenWidth * sizeFraction.x, screenHeight * sizeFraction.y);
164
165 // Apply minimum size constraints
166 calculatedSize.x = std::max(calculatedSize.x, minSize.x);
167 calculatedSize.y = std::max(calculatedSize.y, minSize.y);
168
169 // Apply maximum size constraints
170 calculatedSize.x = std::min(calculatedSize.x, maxSize.x > 0 ? maxSize.x : screenWidth);
171 calculatedSize.y = std::min(calculatedSize.y, maxSize.y > 0 ? maxSize.y : screenHeight);
172
173 return calculatedSize;
174}
