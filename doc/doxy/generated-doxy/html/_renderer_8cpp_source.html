95void Renderer::renderLights(const std::vector<const LightPoint*>& lights,
96 const std::vector<const Transform::Transform*>& lightTransforms,
97 const double timeOfFrame) const {
98 size_t lightCount = static_cast<int>(lights.size());
99 Shader::setUniform("uLights.count", lightCount);
100 for (size_t lightIndex = 0; lightIndex < lightCount; lightIndex++) {
101 const LightPoint& light = *lights[lightIndex];
102 std::string iStr = std::to_string(lightIndex);
103 const Transform::Transform& transform = *lightTransforms[lightIndex];
104 const Transform::Transform& interpolatedTransform =
105 interpolationTransforms.at(&transform).interpolate(static_cast<float>(timeOfFrame));
106
107 Position lightPosViewSpace =
108 Transform::Transformer::transformVector(interpolatedTransform.getPosition(), getView());
109 Shader::setUniform("uLights.posInViewSpace[" + iStr + "]", lightPosViewSpace);
110
111 if (!light.isDirty()) continue;
112
113 Vec3F lightColor = light.getColor().getRGBVec3FNormalized();
114 float lightIntensity = light.getIntensity();
115 float lightRadius = light.getRadius();
116 Shader::setUniform("uLights.color[" + iStr + "]", lightColor);
117 Shader::setUniform("uLights.intensity[" + iStr + "]", lightIntensity);
118
119 Shader::setUniform("uLights.radius[" + iStr + "]", lightRadius);
120 light.setClean();
121 }
122}

180void Renderer::render(double timeOfFrame) {
181 drawCounter.startCounter();
182 const View& viewMat = getView();
183 const Projection& projMat = getProjection();
184 const VP& viewProjMat = getViewProjection();
185 shader.bind(); // Activate the shader program before transform, material and lighting setup
186 frustum.update(viewProjMat);
187
188//#pragma omp parallel for default(shared) schedule(dynamic)
189 for (int meshIndex = 0; meshIndex < meshesToRender.size(); meshIndex++) {
190 const ColorMesh& mesh = *meshesToRender[meshIndex];
191 const Transform::Transform& transform = *meshTransforms[meshIndex];
192 Transform::Transform interpolatedTransform =
193 interpolationTransforms[&transform].interpolate(static_cast<float>(timeOfFrame));
194
195 Model model = interpolatedTransform.getModelMatrix();
196
197 Math::BoundingVolume transformedBoundingVol =
198 Transform::Transformer::transformBoundingVolume(mesh.getBoundingVolume(), interpolatedTransform);
199 {
200 std::lock_guard lockMutex(frustumMutex);
201 isContainedInFrustum.push_back(frustum.contains(transformedBoundingVol));
202 }
203 MVP MVPMat = viewProjMat * model;
204 MV MV = viewMat * model;
205 NormalMat normalMat;
206 normalMat.makeNormalMatrix(MV);
207 {
208 std::lock_guard lockMVP(mvpMutex);
209 mvs.push_back(MV);
210 }
211 {
212 std::lock_guard lockMV(mvMutex);
213 mvps.push_back(MVPMat);
214 }
215 {
216 std::lock_guard lockNormalMat(normalMatMutex);
217 normalMats.push_back(normalMat);
218 }
219 }
220
221 std::string renderedMeshesPtr;
222 for (int i = 0; i < meshesToRender.size(); i++) {
223 if (!isContainedInFrustum[i]) continue;
224 const ColorMesh& mesh = *meshesToRender[i];
225 const Material& material = *meshMaterials[i];
226 applyTransform(mvs[i], mvps[i], normalMats[i]);
227 applyMaterial(material);
228 mesh.sendToGpuBuffers();
229 renderMesh(mesh);
230 renderedMeshesPtr += std::to_string(i) + " ";
231 }
232 renderLights(lights, lightTransforms, timeOfFrame);
233 applySun(sun);
234 applyFog(fog, camera.transform->getPosition());
235 applySkybox(skybox, viewMat, projMat);
236 hasRenderBeenCalled = true;
237}

275void Renderer::sendMeshData(const ColorMesh& mesh, const Material& material, const Transform::Transform& transform) {
276 D_ASSERT_TRUE(!mesh.isBeingBuilt(), "Mesh is being built");
277
278 RenderType renderType = mesh.getRenderType();
279 interpolationTransforms[&transform].pushTransform(transform);
280
281 if (mesh.getVertices().empty()) {
282 Console::warn("Mesh has no vertices");
283 return;
284 }
285
286 if (renderType == RenderType::SingleDrawStatic) {
287 D_ASSERT_TRUE(false, "Static meshes are not supported");
288 return;
289 }
290 if (renderType == RenderType::InstancedStatic) {
291 D_ASSERT_TRUE(false, "Instanced static meshes are not supported");
292 return;
293 }
294 if (renderType == RenderType::InstancedDynamic) {
295 meshesToRender.push_back(&mesh);
296 meshMaterials.push_back(&material);
297 meshTransforms.push_back(&transform);
298 instances[&mesh].push_back(meshTransforms.size());
299 return;
300 }
301 if (renderType == RenderType::SingleDrawDynamic) {
302 meshesToRender.push_back(&mesh);
303 meshMaterials.push_back(&material);
304 meshTransforms.push_back(&transform);
305 return;
306 }
307 D_ASSERT_TRUE(false, "Unknown render type");
308}
