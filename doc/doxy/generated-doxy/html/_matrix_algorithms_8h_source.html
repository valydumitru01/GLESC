397 template <typename Type, size_t N, size_t M, size_t X>
398 static void matrixMatrixMulColMaj(const MatrixData<Type, M, N>& matrix1,
399 const MatrixData<Type, X, M>& matrix2,
400 MatrixData<Type, X, N>& result) {
401 D_ASSERT_NOT_EQUAL(&matrix1, &result, "Cannot multiply matrix in place. Use matrixMatrixMulNaiveInPlace.");
402 D_ASSERT_NOT_EQUAL(&matrix2, &result, "Cannot multiply matrix in place. Use matrixMatrixMulNaiveInPlace.");
403 for (size_t i = 0; i < N; ++i) {
404 for (size_t j = 0; j < X; ++j) {
405 Type sum = static_cast<Type>(0);
406 Type c = static_cast<Type>(0); // A running compensation for lost low-order bits.
407 for (size_t k = 0; k < M; ++k) {
408 Type product = matrix1[k][i] * matrix2[j][k];
409 Type y = product - c;
410 Type t = sum + y;
411 c = (t - sum) - y;
412 sum = t;
413 }
414 result[j][i] = sum;
415 }
416 }
417 }

674 template <typename ModelType, typename NormalMatType>
675 static void calculateNormalMatrix(const MatrixData<ModelType, 4, 4>& MVMat,
676 MatrixData<NormalMatType, 3, 3>& normalMat) {
677 // Calculate the inverse of the 3x3 matrix
678 MatrixData<ModelType, 4, 4> inverseMat;
679 MatrixAlgorithms::inverse4x4(MVMat, inverseMat);

766 template <typename Type, size_t N>
767 static void
768 inverse4x4(const MatrixData<Type, N, N>& matrix, MatrixData<Type, N, N>& result) {
769 Type determinant;
770 MatrixAlgorithms::determinant4x4(matrix, determinant);
771 D_ASSERT_TRUE(MatrixAlgorithms::isInvertible(matrix), "Matrix is not invertible.");
772 D_ASSERT_FALSE(&matrix == &result, "Cannot invert matrix in place.");
773 Type A2323 = matrix[2][2] * matrix[3][3] - matrix[2][3] * matrix[3][2];
774 Type A1323 = matrix[2][1] * matrix[3][3] - matrix[2][3] * matrix[3][1];
775 Type A1223 = matrix[2][1] * matrix[3][2] - matrix[2][2] * matrix[3][1];
776 Type A0323 = matrix[2][0] * matrix[3][3] - matrix[2][3] * matrix[3][0];
777 Type A0223 = matrix[2][0] * matrix[3][2] - matrix[2][2] * matrix[3][0];
778 Type A0123 = matrix[2][0] * matrix[3][1] - matrix[2][1] * matrix[3][0];
779 Type A2313 = matrix[1][2] * matrix[3][3] - matrix[1][3] * matrix[3][2];
780 Type A1313 = matrix[1][1] * matrix[3][3] - matrix[1][3] * matrix[3][1];
781 Type A1213 = matrix[1][1] * matrix[3][2] - matrix[1][2] * matrix[3][1];
782 Type A2312 = matrix[1][2] * matrix[2][3] - matrix[1][3] * matrix[2][2];
783 Type A1312 = matrix[1][1] * matrix[2][3] - matrix[1][3] * matrix[2][1];
784 Type A1212 = matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[2][1];
785 Type A0313 = matrix[1][0] * matrix[3][3] - matrix[1][3] * matrix[3][0];
786 Type A0213 = matrix[1][0] * matrix[3][2] - matrix[1][2] * matrix[3][0];
787 Type A0312 = matrix[1][0] * matrix[2][3] - matrix[1][3] * matrix[2][0];
788 Type A0212 = matrix[1][0] * matrix[2][2] - matrix[1][2] * matrix[2][0];
789 Type A0113 = matrix[1][0] * matrix[3][1] - matrix[1][1] * matrix[3][0];
790 Type A0112 = matrix[1][0] * matrix[2][1] - matrix[1][1] * matrix[2][0];
791
792 Type invDet = 1.0 / determinant;
793
794 result[0][0] = invDet * (matrix[1][1] * A2323 - matrix[1][2] * A1323 + matrix[1][3] * A1223);
795 result[0][1] = invDet * -(matrix[0][1] * A2323 - matrix[0][2] * A1323 + matrix[0][3] * A1223);
796 result[0][2] = invDet * (matrix[0][1] * A2313 - matrix[0][2] * A1313 + matrix[0][3] * A1213);
797 result[0][3] = invDet * -(matrix[0][1] * A2312 - matrix[0][2] * A1312 + matrix[0][3] * A1212);
798 result[1][0] = invDet * -(matrix[1][0] * A2323 - matrix[1][2] * A0323 + matrix[1][3] * A0223);
799 result[1][1] = invDet * (matrix[0][0] * A2323 - matrix[0][2] * A0323 + matrix[0][3] * A0223);
800 result[1][2] = invDet * -(matrix[0][0] * A2313 - matrix[0][2] * A0313 + matrix[0][3] * A0213);
801 result[1][3] = invDet * (matrix[0][0] * A2312 - matrix[0][2] * A0312 + matrix[0][3] * A0212);
802 result[2][0] = invDet * (matrix[1][0] * A1323 - matrix[1][1] * A0323 + matrix[1][3] * A0123);
803 result[2][1] = invDet * -(matrix[0][0] * A1323 - matrix[0][1] * A0323 + matrix[0][3] * A0123);
804 result[2][2] = invDet * (matrix[0][0] * A1313 - matrix[0][1] * A0313 + matrix[0][3] * A0113);

996 template <typename Type, size_t N, size_t M>
997 static GaussianEliminationData<Type, N, M>
998 gaussianElimination(const MatrixData<Type, N, M>& matrix) {
999 MatrixData<Type, N, M * 2> augmentedMatrix;
1000 MatrixData<Type, N, M> identityMatrix;
1001 MatrixData<Type, N, M> inverseResult;
1002
1003 // Initialize identity matrix
1004 MatrixAlgorithms::setMatrixDiagonal(identityMatrix, Type(1));
1005
1006
1007 // Augment the original matrix with identity matrix
1008 for (size_t i = 0; i < N; ++i) {
1009 for (size_t j = 0; j < M; ++j) {
1010 augmentedMatrix[i][j] = matrix[i][j];
1011 augmentedMatrix[i][N + j] = identityMatrix[i][j];
1012 }
1013 }
1014
1015 // Perform Gaussian elimination on the augmented matrix
1016 auto result = gaussianEliminationAlgorithm(augmentedMatrix);
1017
1018 // Extract the inverse matrix from the augmented matrix
1019 for (size_t i = 0; i < N; i++) {
1020 for (size_t j = 0; j < M; j++) {
1021 inverseResult[i][j] = augmentedMatrix[i][j + M];
1022 }
1023 }
1024
1025 // Extract the rank of the matrix
1026 // Rank is the number of non-zero rows in the matrix
1027 unsigned int rank = 0;
1028 for (size_t i = 0; i < N; ++i) {
1029 bool isZeroRow = true;
1030 for (size_t j = 0; j < N; ++j) {
1031 if (augmentedMatrix[i][j] != 0) {
1032 isZeroRow = false;
1033 break;
1034 }
1035 }
1036 if (!isZeroRow)
1037 ++rank;
1038 }
1039
1040 // Extract the determinant of the matrix
1041 // Determinant is the product of diagonal entries of the matrix
1042 // Also, the result must be multiplied by -1 if the number of row swaps is odd
1043 Type determinant = Type(1);
1044 for (size_t i = 0; i < N; ++i) {
1045 determinant *= augmentedMatrix[i][i];
1046 }
1047
1048 bool isRowSwapOdd = result.rowSwaps.size() % 2 == 1;
1049 determinant *= isRowSwapOdd ? Type(-1) : Type(1);
1050
1051 GaussianEliminationData<Type, N, M> extractedData(inverseResult, rank, determinant);
1052 return extractedData;
1053 }

1063 template <class Type, size_t N>
1064 static Type laplaceExpansionDeterminant(const MatrixData<Type, N, N>& matrix) {
1065 if constexpr (N == 1) {
1066 return matrix[0][0];
1067 }
1068 else if constexpr (N == 2) {
1069 Type det;
1070 MatrixAlgorithms::determinant2x2(matrix, det);
1071 return det;
1072 }
1073 else if constexpr (N == 3) {
1074 Type det;
1075 MatrixAlgorithms::determinant3x3(matrix, det);
1076 return det;
1077 }
1078 else if constexpr (N == 4) {
1079 Type det;
1080 MatrixAlgorithms::determinant4x4(matrix, det);
1081 return det;
1082 }
1083 else {
1084 Type result = Type();
1085 Type sign = 1;
1086
1087 for (size_t j = 0; j < N; ++j) {
1088 result += sign * matrix[0][j] * cofactor(matrix, Type(0), j);
1089 sign = -sign; // Alternate sign for cofactor expansion
1090 }
1091 return result;
1092 }

1099 template <typename Type, size_t N, size_t M>
1100 struct GaussianEliminationTrackedResult {
1101 MatrixData<Type, N, M> resultMatrix;
1102 std::map<std::pair<size_t, size_t>, std::pair<size_t, size_t>> rowSwaps;
1103 std::vector<std::tuple<size_t, size_t, Type>> rowMultipleAdditions;
1104 std::vector<std::pair<size_t, Type>> rowMultiplications;

1106
1115 template <typename Type, size_t N, size_t M>
1116 static GaussianEliminationTrackedResult<Type, N, M>
1117 gaussianEliminationAlgorithm(const MatrixData<Type, N, M>& inputMatrix) {
1118 GaussianEliminationTrackedResult<Type, N, M> trackedResult;
1119
1120 // Copy input matrix to result matrix
1121 copyMatrix(trackedResult.resultMatrix, inputMatrix);
1122
1123 // Perform Gaussian elimination with partial pivoting
1124 for (size_t i = 0; i < std::min(N, M); ++i) {
1125 // Search for maximum in this column
1126 size_t maxRow = i;
1127 Type maxElem = std::fabs(trackedResult.resultMatrix[i][i]);
1128 for (size_t k = i + 1; k < N; ++k) {
1129 Type absValue = std::fabs(trackedResult.resultMatrix[k][i]);
1130 if (absValue > maxElem) {
1131 maxElem = absValue;
1132 maxRow = k;
1133 }
1134 }
1135
1136 // Check for zero pivot value (potential singularity)
1137 if (maxElem == Type(0)) {
1138 throw std::runtime_error("Singular matrix, no unique solution exists.");
1139 }
1140
1141 // Swap maximum row with current row (column by column)
1142 if (i != maxRow) {
1143 swapRows(trackedResult.resultMatrix, i, maxRow);
1144 trackedResult.rowSwaps[{i, maxRow}] = {maxRow, i};
1145 }
1146
1147
1148 // Make all rows below this one 0 in current column
1149 for (size_t k = i + 1; k < N; ++k) {
1150 Type c = -trackedResult.resultMatrix[k][i] / trackedResult.resultMatrix[i][i];
1151 for (size_t j = i; j < M; ++j) {
1152 if (i == j) {
1153 trackedResult.resultMatrix[k][j] = 0;
1154 }
1155 else {
1156 trackedResult.resultMatrix[k][j] += c * trackedResult.resultMatrix[i][j];
1157 }
1158 }
1159 trackedResult.rowMultipleAdditions.push_back({k, i, c});
1160 }
1161 }
1162
1163 // Back-substitution (if the system is square and N == M)
1164 if (N == M) {
1165 for (size_t i = N; i-- > 0;) {
1166 Type sum = trackedResult.resultMatrix[i][N];
1167 for (size_t j = i + 1; j < N; ++j) {
1168 sum -= trackedResult.resultMatrix[i][j] * trackedResult.resultMatrix[j][N];
1169 }
1170 trackedResult.resultMatrix[i][N] = sum / trackedResult.resultMatrix[i][i];
1171 trackedResult.rowMultiplications.push_back({i, 1 / trackedResult.resultMatrix[i][i]});
1172 }
1173 }
1174
1175 return trackedResult;
1176 }

1187 template <class Type, size_t N>
1188 static Type cofactor(const MatrixData<Type, N, N>& matrix, size_t row, size_t col) {
1189 MatrixData<Type, N - 1, N - 1> subMatrix;
1190
1191 size_t subI = 0;
1192 for (size_t i = 0; i < N; ++i) {
1193 if (i == row)
1194 continue;
1195
1196 size_t subJ = 0;
1197 for (size_t j = 0; j < N; ++j) {
1198 if (j == col)
1199 continue;
1200
1201 subMatrix[subI][subJ] = matrix[i][j];
1202 subJ++;
1203 }
1204 subI++;
1205 }
1206
1207 return laplaceExpansionDeterminant(subMatrix);
1208 }
