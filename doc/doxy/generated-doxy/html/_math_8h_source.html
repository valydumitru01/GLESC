556 template <typename TypeX, typename TypeMin1, typename TypeMax1, typename TypeMin2, typename TypeMax2>
557 static TypeX remap(TypeX x, TypeMin1 min1, TypeMax1 max1, TypeMin2 min2, TypeMax2 max2) {
558 // Ensure the input value is clamped within the input range
559 x = Math::clamp(x, min1, max1);
560 // Calculate the proportion of x within the input range
561 TypeX t = (x - min1) / (max1 - min1);
562 // Map the proportion to the output range
563 return min2 + t * (max2 - min2);

662 template <typename Type, typename = std::enable_if_t<std::is_arithmetic_v<Type>>>
663 [[nodiscard]] Type generateRandomNumber(Type min, Type max) {
664 S_ASSERT_TRUE(std::is_arithmetic_v<Type>, "Type must be arithmetic");
665 D_ASSERT_TRUE(max >= min, "Max must be greater than min");
666 D_ASSERT_TRUE((max + GLESC::Math::abs(min)) < std::numeric_limits<Type>::max(),
667 "(Max - min) must be less than the max value of the type");
668 static std::random_device rd;
669 static std::mt19937 mt(rd());
670
671 // If the type is uint, we use the fast unsigned int type for the distribution
672 if constexpr (std::is_unsigned_v<Type> && std::is_integral_v<Type>) {
673 std::uniform_int_distribution<std::mt19937::result_type> dist(min, max);
674 return dist(mt);
675 }
676 // If the type is int, we just use the int type for the int distribution
677 else if constexpr (std::is_integral_v<Type>) {
678 std::uniform_int_distribution<Type> dist(min, max);
679 return dist(mt);
680 }
681 else if constexpr (std::is_floating_point_v<Type>) {
682 // For other types (int, etc.), use the type directly
683 std::uniform_real_distribution<Type> dist(min, max);
684 return dist(mt);
685 }
686 else {
687 D_ASSERT_TRUE(false, "Type not supported for random number generation");
688 }
689 }

715 template <typename Type, Type min, Type max>
716 constexpr Type generateCompileTimeRandomNumber() {
717 static_assert(min < max, "max must be greater than min");
718
719 // Extracting compile time in hh:mm:ss format
720 constexpr auto time = __TIME__;
721 constexpr auto hour = (time[0] - '0') * 10 + (time[1] - '0');
722 constexpr auto minute = (time[3] - '0') * 10 + (time[4] - '0');
723 constexpr auto second = (time[6] - '0') * 10 + (time[7] - '0');
724
725 // Calculating a pseudo-random seed from the current time
726 constexpr unsigned seed = hour * 3600 + minute * 60 + second;
727
728 // Parameters for the Linear Congruential Generator
729 constexpr unsigned a = 1664525; // the multiplier
730 constexpr unsigned c = 1013904223; // the increment
731
732 // Modulus is the difference between max and min. Cast to Type to ensure same type arithmetic
733 constexpr Type modulus = static_cast<Type>(max - min);
734
735 // Generating the random number using the Linear Congruential Generator formula
736 // Cast each part of the operation to Type to ensure the arithmetic is done in the correct type
737 constexpr Type randomNumber = static_cast<Type>((a * seed + c) % modulus) + min;
738
739 return randomNumber;
740 }
